```bash
i@gros-44:~/embedded-sys_energy$ cat /sys/devices/system/cpu/cpu0/cpufreq/cpuinfo_max_freq
3900000
i@gros-44:~/embedded-sys_energy$ echo $freq | sudo-g5k tee /sys/devices/system/cpu/cpu*/cpufreq/scaling_max_freq
3900000
```
